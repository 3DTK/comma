152 class to_format
153 {
154  public:
155  to_format( const std::string& fields = "", bool full_xpath = true )
156  : full_xpath_( full_xpath )
157  , first_( true )
158  {
159  if( fields.empty() ) { return; }
160  std::vector< std::string > v = comma::split( fields, ',' );
161  for( std::size_t i = 0; i < v.size(); ++i ) { fields_.insert( v[i] ); }
162  }
163 
164  template < typename K, typename T >
165  void apply( const K& name, const T& value )
166  {
167  append( name );
168  visiting::do_while< !boost::is_fundamental< T >::value
169  && !boost::is_same< T, std::string >::value
170  && !boost::is_same< T, boost::posix_time::ptime >::value >::visit( name, value, *this );
171  trim( name );
172  }
173 
174  template < typename K, typename T >
175  void apply_next( const K& name, const T& value ) { comma::visiting::visit( name, value, *this ); }
176 
177  template < typename K, typename T >
178  void apply_final( const K& key, const T& value )
179  {
180  //std::cerr << "---> fields_.size() = " << fields_.size() << std::endl;
181  (void)key;
182  if( !fields_.empty() )
183  {
184  if( full_xpath_ )
185  {
186  bool found = false; // quick and dirty, should be fast and is called only once anyway
187  for( std::set< std::string >::const_iterator it = fields_.begin(); !found && it != fields_.end(); ++it ) // quick and dirty
188  {
189  found = xpath_ <= comma::xpath( *it );
190  }
191  if( !found ) { return; }
192  }
193  else
194  {
195  if( fields_.find( xpath_.elements.back().to_string() ) == fields_.end() ) { return; }
196  }
197  }
198  if( !first_ ) { format_ += ','; }
199  else { first_ = false; }
200  format_ += format::value_impl( value );
201  }
202 
203  const std::string& operator()() const { return format_; }
204 
205  private:
206  std::set< std::string > fields_;
207  bool full_xpath_;
208  std::string format_;
209  xpath xpath_;
210  bool first_;
211  const xpath& append( std::size_t index ) { xpath_.elements.back().index = index; return xpath_; }
212  const xpath& append( const char* name ) { xpath_ /= xpath::element( name ); return xpath_; }
213  const xpath& trim( std::size_t ) { xpath_.elements.back().index = boost::optional< std::size_t >(); return xpath_; }
214  const xpath& trim( const char* ) { xpath_ = xpath_.head(); return xpath_; }
215 };
