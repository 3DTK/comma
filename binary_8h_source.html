56 template < typename S >
57 inline binary< S >::binary( const std::string& f, const std::string& column_names, bool full_path_as_name, const S& sample )
58  : format_( f == "" ? csv::format::value( sample ) : f )
59 {
60  if( format_.size() == sizeof( S ) && format_.string() == csv::format::value( sample ) && join( csv::names( column_names, full_path_as_name, sample ), ',' ) == join( csv::names( full_path_as_name ), ',' ) ) { return; }
61  binary_ = impl::binary_visitor( format_, join( csv::names( column_names, full_path_as_name, sample ), ',' ), full_path_as_name );
62  visiting::apply( *binary_, sample );
63 }

65 template < typename S >
66 inline binary< S >::binary( const options& o, const S& sample )
67  : format_( o.format().string() == "" ? csv::format::value( sample ) : o.format().string() )
68 {
69  if( format_.size() == sizeof( S ) && format_.string() == csv::format::value( sample ) && join( csv::names( o.fields, o.full_xpath, sample ), ',' ) == join( csv::names( o.full_xpath ), ',' ) ) { return; }
70  binary_ = impl::binary_visitor( format_, join( csv::names( o.fields, o.full_xpath, sample ), ',' ), o.full_xpath );
71  visiting::apply( *binary_, sample );
72 }

74 template < typename S >
75 inline const S& binary< S >::get( S& s, const char* buf ) const
76 {
77  if( binary_ )
78  {
79  impl::frobinary_ f( binary_->offsets(), binary_->optional(), buf );
80  visiting::apply( f, s );
81  }
82  else // quick and dirty for better performance
83  {
84  ::memcpy( reinterpret_cast< char* >( &s ), buf, sizeof( S ) );
85  }
86  return s;
87 }
