306 template < typename S >
307 inline const S* ascii_input_stream< S >::read()
308 {
309  while( is_.good() && !is_.eof() )
310  {
312  std::string s;
313  std::getline( is_, s );
314  if( !s.empty() && *s.rbegin() == '\r' ) { s = s.substr( 0, s.length() - 1 ); } // windows... sigh...
315  if( s.empty() ) { continue; }
316  result_ = default_;
317  line_ = split
318 ( s, ascii_.delimiter() );
319  ascii_.get( result_, line_ );
320  return &result_;
321  }
322  return NULL;
323 }

425 template < typename S >
426 inline const S* binary_input_stream< S >::read()
427 {
428  while( true ) // reading a big chunk for better performance
429  {
430  if( ready() )
431  {
432  result_ = default_;
433  binary_.get( result_, cur_ );
434  last_ = cur_;
435  cur_ += binary_.format().size();
436  offset_ -= binary_.format().size();
437  if( cur_ >= end_ ) { cur_ = begin_; offset_ = 0; }
438  return &result_;
439  }
440  bool bad = is_.eof() || !is_.good() || is_.bad() || is_.fail();
441  if( offset_ > 0 && bad ) { COMMA_THROW( comma::exception, "expected at least " << binary_.format().size() << " bytes; got " << offset_ ); }
442  if( bad ) { return NULL; }
443  std::size_t size = end_ - cur_ - offset_;
444  // this is a painful part to read only the available bytes
445  // if using sockets or pipes, make sure that data is there
446  // e.g. using ::select(), comma::Io::Select(), or alike
447  // on the internet they say there is no good way,
448  // (google "readsome does not work"
449  // STL streams seem to take you to a world of pain...)
450  std::streamsize a = is_.rdbuf()->in_avail();
451  if( a < 0 ) { continue; }
452  std::size_t available = a;
453  if( size > available )
454  {
455  size = offset_ + available > binary_.format().size()
456  ? available
457  : binary_.format().size() - offset_;
458 
459  //if( binary_.format().size() < available )
460  //{
461  // size = available;
462  //}
463  //else
464  //{
465  // size = size % binary_.format().size();
466  // if( size == 0 ) { size = binary_.format().size(); }
467  //}
468  }
469  is_.read( cur_ + offset_, size ); // blocks till full size bytes read
470  std::streamsize count = is_.gcount();
471  if( count < 0 ) { continue; }
472  offset_ += count;
473  }
474 }
