37 template < unsigned int Size, bool Signed >
38 struct little_endian_int : public packed::field< little_endian_int< Size, Signed >, typename little_endian_traits< Size, Signed >::type, Size >
39 {
40  static const unsigned int size = Size;
41 
42  typedef typename little_endian_traits< Size, Signed >::type type;
43 
44  BOOST_STATIC_ASSERT( size <= sizeof( type ) );
45 
46  typedef packed::field< little_endian_int< Size, Signed >, typename little_endian_traits< Size, Signed >::type, Size > base_type;
47 
48  static type default_value() { return 0; }
49 
50  static void pack( char* storage, type value )
51  {
52  for( unsigned int i = 0; i < size; ++i, value >>= 8 ) { storage[i] = value & 0xff; }
53  }
54 
55  static type unpack( const char* storage )
56  {
57  type v = 0;
58  int shift = 0;
59  unsigned int i = 0;
60  for( ; i < size; ++i, shift += 8 )
61  {
62  v += ( unsigned char )( storage[i] ) << shift;
63  }
64  if( storage[ size - 1 ] & 0x80 )
65  {
66  for( ; i < sizeof( type ); ++i, shift += 8 ) { v += 0xff << shift; }
67  }
68  return v;
69  }
70 
71  const little_endian_int& operator=( const little_endian_int& rhs ) { return base_type::operator=( rhs ); }
72 
73  const little_endian_int& operator=( type rhs ) { return base_type::operator=( rhs ); }
74 };
