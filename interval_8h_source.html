127 template < typename T >
128 inline T mod( T t, const interval< T >& r )
129 {
130  T size = r.size();
131  if( math::less( t, r().first ) )
132  {
133  return r().second - impl::mod( r().second - t, size );
134  }
135  return r().first + impl::mod( t - r().first, size );
136 }
