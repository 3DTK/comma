161 template < typename T >
162 inline void cyclic_buffer< T >::push( const T& t )
163 {
164  if( size() == vector_.size() ) { COMMA_THROW( comma::exception, "full" ); }
165  vector_[ end_() ] = t;
166  ++end_;
167  empty_ = false;
168 }

177 template < typename T >
178 inline void cyclic_buffer< T >::pop( std::size_t n )
179 {
180  if( empty() ) { return; }
181  if( n >= size() ) { clear(); return; }
182  begin_ += n;
183  empty_ = begin_ == end_;
184 }
