97 template < typename S >
98 inline const std::vector< std::string >& ascii< S >::put( const S& s, std::vector< std::string >& v ) const
99 {
100  if( v.empty() ) { v.resize( ascii_.size() ); }
101  impl::to_ascii f( ascii_.indices(), v );
102  if( precision_ ) { f.precision( *precision_ ); }
103  visiting::apply( f, s );
104  return v;
105 }

107 template < typename S >
108 inline const std::string& ascii< S >::put( const S& s, std::string& line ) const
109 {
110  std::vector< std::string > v;
111  if( !line.empty() ) { v = split( line, delimiter_ ); }
112  line = join( put( s, v ), delimiter_ );
113  return line;
114 }
