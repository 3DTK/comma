84 inline static void init( comma::csv::options& csvoptions, const comma::command_line_options& options, const std::string& defaultFields )
85 {
86  csvoptions.full_xpath = options.exists( "--full-xpath" );
87  csvoptions.fields = options.value( "--fields", defaultFields );
88  if( options.exists( "--binary" ) )
89  {
90  boost::optional< std::string > format = options.optional< std::string >( "--binary" );
91  if( format )
92  {
93  csvoptions.format( options.value< std::string >( "--binary" ) );
94  }
95  }
96  csvoptions.precision = options.value< unsigned int >( "--precision", 6 );
97  csvoptions.delimiter = options.exists( "--delimiter" ) ? options.value( "--delimiter", ',' ) : options.value( "-d", ',' );
98 }

114 inline std::string options::usage()
115 {
116  std::ostringstream oss;
117  oss << " --binary,-b <format> : use binary format" << std::endl;
118  oss << " --delimiter,-d <delimiter> : default: ','" << std::endl;
119  oss << " --fields,-f <names> : field names, e.g. t,,x,y,z" << std::endl;
120  oss << " --full-xpath : expect full xpaths as field names" << std::endl;
121  oss << " --precision <precision> : floating point precision; default: 6" << std::endl;
122  oss << format::usage();
123  return oss.str();
124 }
