64  const cyclic& operator+=( const T& t )
65  {
66  value_ = math::mod( value_ + t, interval_ );

68  }
69  const cyclic& operator-=( const T& t )
70  {
71  T s = math::mod( t, interval_ );
72  value_ = ( math::less( value_, s ) ? interval_().second : interval_().first ) + value_ - s;
73  return *this;
74  }

96 template < typename T >
97 bool cyclic< T >::between( const cyclic< T >& lower, const cyclic< T >& upper ) const
98 {
99  if( less( upper.value_, lower.value_ ) )
100  {
101  return !less( value_, lower.value_ ) || less( value_, upper.value_ );
102  }
103  else
104  {
105  return !less( value_, lower.value_ ) && less( value_, upper.value_ );
106  }
