87  {
88  if( !( path_ <= branch_ ) ) { return; } // visit, only if on the branch
89  append_( key );
90  boost::optional< const boost::property_tree::ptree& > t = ptree_.get_child_optional( path_.to_string( '.' ) );
91  if( !permissive_ || t )
92  {
93  if( !t ) { COMMA_THROW( comma::exception, "path " << path_.to_string( '.' ) << " not found" ); }
94  value.resize( t->size() );

272 inline static void ptree_to_path_value_string_impl( std::ostream& os, boost::property_tree::ptree::const_iterator i, bool is_begin, xpath& path, char equalSign, char delimiter, const std::string& root )
273 {

342 inline boost::property_tree::ptree property_tree::from_name_value_string( const std::string& s, char equalSign, char delimiter )
343 {
344  boost::property_tree::ptree ptree;
345  bool escaped = false;
346  bool quoted = false;
347  std::ostringstream oss;
348  for( std::size_t i = 0; i < s.length(); ++i )
349  {
350  char c = s[i];
351  bool space = false;
352  if( escaped )
353  {
354  escaped = false;
355  }
356  else
357  {
358  switch( c )
359  {
360  case '\\':
361  escaped = true;
362  break;
363  case '"':
364  quoted = !quoted;
365  break;
366  case '{':
367  case '}':
368  space = !quoted;
369  default:
370  if( quoted ) { break; }
371  if( c == equalSign ) { c = ' '; }
372  else if( c == delimiter ) { c = ' '; }
373  break;
374  }
375  }
376  if( space ) { oss << ' '; }
377  oss << c;
378  if( space ) { oss << ' '; }
379  }
380  std::istringstream iss( oss.str() );
381  boost::property_tree::read_info( iss, ptree );
382  return ptree;
383 }

385 inline boost::property_tree::ptree property_tree::from_path_value_string( const std::string& s, char equalSign, char delimiter )
386 {
387  boost::property_tree::ptree ptree;
388  std::vector< std::string > v = comma::split( s, delimiter );
389  for( std::size_t i = 0; i < v.size(); ++i )
390  {
391  if( v[i].empty() ) { continue; }
392  std::vector< std::string > pair = comma::split( v[i], equalSign );
393  if( pair.size() != 2 ) { COMMA_THROW( comma::exception, "expected '" << delimiter << "'-separated xpath" << equalSign << "value pairs; got \"" << v[i] << "\"" ); }
394  ptree.put( boost::property_tree::ptree::path_type( comma::strip( pair[0], '"' ), '/' ), comma::strip( pair[1], '"' ) );
395  }
396  return ptree;
397 }
