107 template < typename T >
108 inline std::vector< T > command_line_options::values( const std::string& name ) const
109 {
110  std::vector< T > v;
111  std::vector< std::string > names = comma::split( name, ',' );
112  for( std::size_t i = 0; i < names.size(); ++i )
113  {
114  typename map_type_::const_iterator it = map_.find( names[i] );
115  if( it == map_.end() ) { continue; }
116  for( std::size_t j = 0; j < it->second.size(); ++j ) { v.push_back( lexical_cast_< T >( it->second[j] ) ); }
117  }
118  return v;
119 }
